## 문제 설명

[문제 보러가기](https://leetcode.com/problems/contains-duplicate/)

배열 안에 중복된 요소가 있는지 검사하는 문제이다.

배열에서 중복된 요소가 있으면 true를, 없다면 false를 return하면 된다.

## Example

```
Input: nums = [1,2,3,1]
Output: true

Input: nums = [1,2,3,4]
Output: false

Input: nums = [1,1,1,3,3,4,3,2,4,2]
Output: true

```

## 풀이

```js

// * @param {number[]} nums
// * @return {boolean}


// 내 풀이 1
var containsDuplicate = function(nums) {
    let count = 0
    for(let i = 0; i < nums.length ; i++){
        for(let j = i+1; j < nums.length; j++ ){
            if(nums[i] === nums[j]) {
                count++
                return count >= 1 //중복 발견하면 바로 return
            }
        }
    }

    return false //중복 없으면 for문 다 돌기 때문에 return false
};

// 내 풀이 2
var containsDuplicate = function(nums) {
    let hash = {} //해시테이블 설정

        hash[num] = true //없으면 해시테이블에 저장

    }

    return false
};
```
